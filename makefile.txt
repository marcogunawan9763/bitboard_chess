# Compiler
CXX = g++
CXXFLAGS = -Wall -Wextra -O2 -Iinclude

# Source files and object files
SRC = src/main.cpp src/bitboard.cpp src/board.cpp src/movegen.cpp
OBJ = $(SRC:src/%.cpp=build/%.o)

# Output binary
TARGET = build/chess_engine

# Default target
all: $(TARGET)

# Rule to link the final executable
$(TARGET): $(OBJ)
	@mkdir -p build
	$(CXX) $(CXXFLAGS) $(OBJ) -o $(TARGET)

# Rule to compile individual .cpp files into .o files
build/%.o: src/%.cpp include/%.h
	@mkdir -p build
	$(CXX) $(CXXFLAGS) -c $< -o $@

# Clean compiled files
clean:
	rm -rf build

# Run the program
run: all
	./build/chess_engine
